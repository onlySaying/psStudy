#include<iostream>
#include<vector>

using namespace std;

int mat[51][51];

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	
	int n = 0;
	int m = 0;
	cin >> n >> m;
	
	for (int i = 0; i < n; i++)
	{
		string temp;
		cin >> temp;
		for (int j = 0; j < m; j++)
		{
			mat[i][j] = temp.at(j) - '0';
		}
	}
	int ans = 1;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			int ver = mat[i][j];
			for (int k = j+1; k < m; k++)
			{
				if (ver == mat[i][k])
				{
					if ((i + k - j) >= n) { continue;}
					
					if (mat[i + k - j][j] == ver)
					{
						if (mat[i + k - j][k] == ver)
						{
							int lineL = k - j + 1;
							lineL *= lineL;
							ans = max(ans, lineL);
						}
					}
				}
			}
		}

	}

	cout << ans;
	
	
}
