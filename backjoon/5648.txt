#include <iostream>
#include <algorithm>
#include<vector>
#include<string>
#include<map>

using namespace std;


int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	int n;
	cin >> n;

	map<long long, int> m;
	string ele;
	long long des;
	long long temp;
	for (int i = 0; i < n; i++)
	{
		cin >> ele;
		des = 0;
		for (int j = 0; j < ele.size(); j++)
		{
			temp = ele[j] - '0';
			for (int k = 0; k < j; k++)
			{
				temp *= 10;
			}
			des += temp;
		}
		m[des]++;
	}


	for (auto iter = m.begin(); iter != m.end(); ++iter )
	{
		for (int i = 0; i < iter->second; i++)
		{
			cout << iter->first << "\n";
		}
	}

	return 0;
}
