#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

typedef long long ll;

struct cmp
{
    bool operator()(int a, int b)
    {
        if (abs(a) == abs(b))
        {
            return a > b;
        }
        return abs(a) > abs(b);
    }
};

int main(void) 
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;
    priority_queue<int, vector<int>, cmp> pq;
    int ele;
    for (int i = 0; i < n; i++)
    {
        cin >> ele;
        if (ele != 0)
        {
            pq.push(ele);
        }
        else
        {
            if (pq.empty()) { cout << "0\n"; }
            else 
            {
                cout << pq.top() << '\n';
                pq.pop();
            }

        }
    }
   
     return 0;
}
