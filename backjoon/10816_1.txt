#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int n;
vector<int> v;

int main(void) 
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    cin >> n;
    int ele;

    for (int i = 0; i < n; i++)
    {
        cin >> ele;
        v.push_back(ele);
    }
    int m;
    cin >> m;
    sort(v.begin(), v.end());
    while (m--)
    {
        int target;
        cin >> target;
        cout << upper_bound(v.begin(),v.end(), target) - lower_bound(v.begin(), v.end(), target) << " ";
    }
    
    return 0;
}
