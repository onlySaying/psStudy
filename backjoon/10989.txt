#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <tuple>
using namespace std;

int num[10001];

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;
	int temp;
	int max = 0;
	for (int i = 0; i < n; i++)
	{
		cin >> temp;
		num[temp]++;
		if (max < temp)
		{
			max = temp;
		}
	}

	for (int i = 1; i <= max; i++)
	{
		if (num[i] == 0) { continue; }
		while (num[i]--)
		{
			cout << i << "\n";
		}
	}
}
